	public int numWays(int n) {
        if(n==0) return 1;
        if(n<=2) return n;
        //动态规划
        int[] dp=new int[n+1];
        dp[1]=1;
        dp[2]=2;
        //在第n阶可由n-1阶跳一步，和在第n-2阶跳两步的可能跳法组成
        for(int i=3;i<=n;i++){
            dp[i]=(dp[i-1]+dp[i-2])%1000000007;
        }
        return dp[n];
    }
