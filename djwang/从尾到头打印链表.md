	public int[] reversePrint(ListNode head) {
       //压栈出栈
	   //时间空间复杂度为O(n)
        int MAXSIZE=15000;
        int[] stack = new int[MAXSIZE];
        int size=0;
        while(head!=null){
            stack[size++]=head.val;
            head=head.next;
        };
        int[] value=new int[size];
        for(int i=0;i<size;i++)
            value[i]=stack[size-i-1];
        return value;
    }
